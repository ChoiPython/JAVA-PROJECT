create database test;

use test;

create table user(
사원번호 INT(10) not null primary key,
사원이름 CHAR(12),
부서 CHAR(12),
직급 CHAR(12),
포인트 INT(5),
사진 CHAR(150)
);


create table Attendance(
날짜 date,
사원번호 int,
상태 char(10),
primary key(날짜,사원번호),
foreign key(사원번호) references user(사원번호)
);

// 상점 데이터 - 사원번호(개인키) - 거래내역을 위함
create table trade(
사원번호 int primary key,
반차구매수 int default 0,
뽑기구매수 int default 0,
가격 int default 0,
foreign key(사원번호) references user(사원번호));

// user 테이블 사원번호 전체 추가
 insert into trade(사원번호) select 사원번호 from user;
